-- TYPE TABLE
DECLARE
  TYPE_TABLE_NAME APPS.WSH_DELIVERY_DETAILS_PUB.ID_TAB_TYPE;
BEGIN
  FOR j IN 1..5
    LOOP
      TYPE_TABLE_NAME(j) := j*5487;
    END LOOP; 
 
  FOR indx IN NVL (TYPE_TABLE_NAME.FIRST, 0)..NVL (TYPE_TABLE_NAME.LAST, -1)
  LOOP
    DBMS_OUTPUT.PUT_LINE(TYPE_TABLE_NAME(indx));
  END LOOP;
END;
